// รหัส Google Apps Script (Code.gs) const SHEET_ID = "YOUR_SHEET_ID"; const LINE_ACCESS_TOKEN = "YOUR_LINE_CHANNEL_ACCESS_TOKEN"; function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === "submitRepair") { return saveRepairData(data); } else if (action === "updateStatus") { return updateRepairStatus(data); } } function saveRepairData(data) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName("Repairs"); const caseId = "REQ-" + new Date().getTime(); sheet.appendRow([ caseId, new Date(), data.userId, data.realName, data.licensePlate, data.mileage, data.symptoms, data.estimatedCost, data.location, data.photoUrl, "รอดำเนินการ" ]); sendLineToAdmin(data); // ส่งแจ้งเตือนหา Admin return ContentService.createTextOutput(JSON.stringify({status: "success", caseId: caseId})).setMimeType(ContentService.MimeType.JSON); } function sendLineToAdmin(data) { const message = { "to": "ADMIN_USER_ID_OR_GROUP", "messages": [{ "type": "text", "text": `🚗 มีการแจ้งซ่อมรถยนต์ใหม่!\n👤 ผู้แจ้ง: ${data.realName}\n🔢 ทะเบียน: ${data.licensePlate}\n🛣️ เลขไมล์: ${data.mileage}\n🔧 อาการ: ${data.symptoms}\n💰 ประเมิน: ${data.estimatedCost} บาท\n📍 สถานที่: ${data.location}` }] }; UrlFetchApp.fetch("https://api.line.me/v2/bot/message/push", { "method": "post", "headers": { "Authorization": "Bearer " + LINE_ACCESS_TOKEN, "Content-Type": "application/json" }, "payload": JSON.stringify(message) }); }
